CSV to HTML Table

Our goal for today is to construct a set of javascript functions to take a CSV file as input and output an HTML table format of that file.

Note: any modifications to read.js in the future will result in changes to this page. If the code blocks below need changes, please consult the source code here.

Reading Text Files as Strings

function readTextFile(file) {
    var rawFile = new XMLHttpRequest();
    var allText; // define where to store the data
    rawFile.open("GET", file, false);
    rawFile.onreadystatechange = function () {
        if (rawFile.readyState === 4) {
            if (rawFile.status === 200 || rawFile.status == 0) {
                allText = rawFile.responseText;
            }
        }
    };
    rawFile.send(null);
    return allText;
}

Reading Text Files as Arrays

function textFileAsArray(file){
    var allText = readTextFile(file);
    var lines = allText.split("\n");
    lines.pop();
    return lines;
}

Converting Text File Arrays to HTML Tables

function csvTable(file,id){
  const lines = textFileAsArray(file);
    // get the table
    var table = document.getElementById(id);
    var newTable = '\n';
    // for each line in the csv
    for (let i = 0; i < lines.length; i++) {
      newTable = newTable.concat("\n");
      // for each comma sep value in the line
      lines[i].split(",").forEach((value) => {
        // if we are at the head of the table
        if (i == 0) {
          newTable = newTable.concat("\n");
        } 
        else if(value[0] == "."){
          console.log(value);
          newTable = newTable.concat('\n');
        }
        else {
          newTable = newTable.concat("\n");
        }
      });
      newTable = newTable.concat("\n");
    }
    newTable = newTable.concat("
" + value + " Link  " + value + "
\n");
    // console.log(newTable);
    table.innerHTML = newTable;
}
